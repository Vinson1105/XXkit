# XXdMap - 一个以路径key存放value的Map<String,String>
map[1][2][3] = 123
	1/2/3 : 123

map[1][2][4](0) = 124A
	1/2/3 : 123
	1/2/4 : (),1,0
	1/2/4/(0) = 124A

map[1][2][4](1) = 124B
	1/2/3 : 123
	1/2/4 : (),2,0,1
	1/2/4/(0) : 124A
	1/2/4/(1) : 124B

map[1][2][4](2) = 124C
	1/2/3 : 123
	1/2/4 : (),3,0,1,2
	1/2/4/.0 : 124A
	1/2/4/.1 : 124B
	1/2/4/.2 : 124C

map[1][2][4](2)[Name] = ccc
	1/2/3 : 123
	1/2/4 : (),3,0,1,2
	1/2/4/.0 : 124A
	1/2/4/.1 : 124B
	1/2/4/.2 : 124C <--- 删除
	1/2/4/.2/Name : ccc

 - []
    不能以‘.’开头

 - ()
    只能是非负数整型
