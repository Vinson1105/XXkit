# XXdMap - 一个以路径key存放value的Map<String,String>
map[1][2][3] = 123
	1/2/3 : 123

map[1][2][4](0) = 124A
	1/2/3 : 123
	1/2/4 : .,1,0
	1/2/4/(0) = 124A

map[1][2][4](1) = 124B
	1/2/3 : 123
	1/2/4 : .,2,0,1
	1/2/4/(0) : 124A
	1/2/4/(1) : 124B

map[1][2][4](2) = 124C
	1/2/3 : 123
	1/2/4 : .,3,0,1,2
	1/2/4/.0 : 124A
	1/2/4/.1 : 124B
	1/2/4/.2 : 124C

map[1][2][4](2)[Name] = ccc
	1/2/3 : 123
	1/2/4 : .,3,0,1,2
	1/2/4/.0 : 124A
	1/2/4/.1 : 124B
	1/2/4/.2 : 124C <--- 删除
	1/2/4/.2/Name : ccc

# 对于Key的限制
 - []
    不能以‘.’开头

 - ()
    只能是非负数整型

# 疑问
- 当执行
map["Class"](0)["Name"] = "AAA";
- 产生的路径为
Class/.0/Name
- 需要完成以下步骤
1. 路径拆解，并逐层合并，查找是否包含数组路径
	如逐一判断：Class、Class/.0、Class/.0/Name
2. 发现Class/.0是一个数组路径，判断标识是'.'
3. 需要在路径Class对应的值中，查找是否有数组标识
	如果'.x'中的数值
		小于数组数量，则直接操作数组项
		等于数组数量+1，则直接追加一个数组项
		大于数组数量+1，则视为不正确操作
